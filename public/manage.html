create table teams ( id serial primary key, name text unique not null, total numeric default 0, show boolean default true ); create table donations ( id serial primary key, team_id integer references teams(id), team_name text, donor_name text, description text, amount numeric not null, created_at timestamptz default now() ); alter table teams enable row level security; alter table donations enable row level security; create policy "public read teams" on teams for select using (true); create policy "public write teams" on teams for all using (true); create policy "public read donations" on donations for select using (true); create policy "public write donations" on donations for all using (true); -- Keep teams.total correct & concurrency-safe (the app never writes it directly) create or replace function apply_donation_to_team() returns trigger language plpgsql as $$ begin if (tg_op = 'INSERT') then update teams set total = coalesce(total,0) + new.amount where id = new.team_id; elsif (tg_op = 'DELETE') then update teams set total = greatest(0, coalesce(total,0) - old.amount) where id = old.team_id; end if; return null; end; $$; create trigger trg_donation_insert after insert on donations for each row execute function apply_donation_to_team(); create trigger trg_donation_delete after delete on donations for each row execute function apply_donation_to_team();

Make totals concurrency-safe — run once

Adds the trigger that keeps teams.total correct when several people record donations at once, then recomputes existing totals from the donation rows to fix any past drift.

create or replace function apply_donation_to_team() returns trigger language plpgsql as $$ begin if (tg_op = 'INSERT') then update teams set total = coalesce(total,0) + new.amount where id = new.team_id; elsif (tg_op = 'DELETE') then update teams set total = greatest(0, coalesce(total,0) - old.amount) where id = old.team_id; end if; return null; end; $$; drop trigger if exists trg_donation_insert on donations; create trigger trg_donation_insert after insert on donations for each row execute function apply_donation_to_team(); drop trigger if exists trg_donation_delete on donations; create trigger trg_donation_delete after delete on donations for each row execute function apply_donation_to_team(); -- recompute existing totals from the donation rows update teams t set total = coalesce(( select sum(d.amount) from donations d where d.team_id = t.id), 0);
